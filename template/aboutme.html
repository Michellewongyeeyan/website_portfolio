shooting game (click the picture below to download exe.file)

                    
                        import pygame
                        import sys
                        import random
                        import math
                        
                        
                        
                        pygame.init()
                        pygame.mixer.init()
                        
                        
                        WIDTH, HEIGHT = 800, 600
                        FPS = 60
                        BULLET_SPEED = 8
                        ERRORCODE_SPEED = 5
                        
                        
                        screen = pygame.display.set_mode((WIDTH, HEIGHT))
                        pygame.display.set_caption("Simple Shooting Game")
                        clock = pygame.time.Clock()
                        
                        
                            
                        
                        player_image = pygame.image.load("../img/player.png").convert()
                        bullet_image = pygame.image.load("../img/bullet.png").convert()
                        errorcode_image = pygame.image.load("../img/Errorcode.png").convert()
                        pygame.display.set_icon(player_image)
                        
                        
                        player_image.set_colorkey((0, 0, 0))
                        bullet_image.set_colorkey((0, 0, 0))
                        errorcode_image.set_colorkey((0, 0, 0))
                        
                        
                        player_image = pygame.transform.scale(player_image, (100, 100))
                        
                        
                        errorcode_image = pygame.transform.scale(errorcode_image, (100, 80))
                        
                        
                        player_rect = player_image.get_rect(center=(WIDTH // 2, HEIGHT // 2))
                        
                        
                        bullets = []
                        
                        
                        errorcodes = []
                        
                        
                        score = 0
                        
                        
                        font = pygame.font.Font(None, 36)
                        small_font = pygame.font.Font(None, 24)
                        
                        
                        congratulations_text = small_font.render(
                            "Yeah! You've overcome all obstacles to your programming education!",
                            True, (255, 255, 255)
                        )
                        congratulations_rect = congratulations_text.get_rect(center=(WIDTH // 2, HEIGHT // 2))
                        
                        
                        instructions_text = small_font.render(
                            "Move your mouse to control the player. Left-click to shoot."
                            "Avoid rocks and shoot them down to win!",
                            True, (255, 255, 255)
                        )
                        instructions_rect = instructions_text.get_rect(topleft=(10, 10))
                        
                        # Start button
                        start_button_text = small_font.render("Click here to start", True, (255, 255, 255))
                        start_button_rect = start_button_text.get_rect(center=(WIDTH // 2, HEIGHT // 2 + 50))
                        
                        # Retry button
                        retry_button_text = small_font.render("Click here to retry.", True, (255, 255, 255))
                        retry_button_rect = retry_button_text.get_rect(center=(WIDTH // 2, HEIGHT // 2 + 50))
                        
                        
                        retry_button_rect.inflate_ip(50, 50)
                        
                        def reset_game():
                            global player_rect, bullets, errorcodes, score
                            player_rect = player_image.get_rect(center=(WIDTH // 2, HEIGHT // 2))
                            bullets = []
                            errorcodes = []
                            score = 0
                        
                        # Game loop
                        running = False
                        frame = 0
                        while not running:
                            for event in pygame.event.get():
                                if event.type == pygame.QUIT:
                                    running = True
                                elif event.type == pygame.MOUSEBUTTONDOWN and event.button == 1:
                                    # Check if the mouse click is inside the start button
                                    if start_button_rect.collidepoint(event.pos):
                                        reset_game()
                                        running = True
                        
                            # Draw everything
                            screen.fill((0, 0, 0))
                            screen.blit(player_image, player_rect)
                            screen.blit(start_button_text, start_button_rect.topleft)
                            screen.blit(instructions_text, instructions_rect)  # Draw instructions on top
                            pygame.display.flip()
                            clock.tick(FPS)
                        
                        # Main game loop
                        running = True
                        game_over = False
                        while running:
                            for event in pygame.event.get():
                                if event.type == pygame.QUIT:
                                    running = False
                        
                            if not game_over:
                        
                                mouse_x, mouse_y = pygame.mouse.get_pos()
                        
                        
                                player_rect.centerx = mouse_x
                                player_rect.centery = mouse_y
                        
                         
                                mouse_click = pygame.mouse.get_pressed()
                                if mouse_click[0] == 1:
                                    bullet_rect = bullet_image.get_rect(center=player_rect.center)
                                    bullets.append(bullet_rect)
                        
                        
                                bullets = [bullet.move(0, -BULLET_SPEED) for bullet in bullets]
                        
                             
                                if random.randint(0, 100) < 5:
                                    errorcode_rect = errorcode_image.get_rect(
                                        center=(random.randint(0, WIDTH), 0)
                                    )
                                    errorcodes.append(errorcode_rect)
                        
                          
                                errorcodes = [errorcode.move(0, ERRORCODE_SPEED) for errorcode in errorcodes]
                        
                        
                                for bullet in bullets[:]:
                                    for errorcode in errorcodes[:]:
                                        if bullet.colliderect(errorcode):
                                            bullets.remove(bullet)
                                            errorcodes.remove(errorcode)
                                            score += 1
                        
                        
                                errorcodes = [errorcode for errorcode in errorcodes if errorcode.bottom < HEIGHT]
                        
                            
                                for errorcode in errorcodes:
                                    if player_rect.colliderect(errorcode):
                                        game_over = True
                        
                           
                                r = int(math.sin(frame * 0.01) * 127 + 128)
                                g = int(math.sin(frame * 0.02) * 127 + 128)
                                b = int(math.sin(frame * 0.03) * 127 + 128)
                                frame += 1
                        
                        
                                screen.fill((r, g, b))
                                screen.blit(player_image, player_rect)
                                for bullet in bullets:
                                    screen.blit(bullet_image, bullet)
                                for errorcode in errorcodes:
                                    screen.blit(errorcode_image, errorcode)
                        
                          
                                score_text = font.render(f"Score: {score}", True, (255, 255, 255))
                                screen.blit(score_text, (10, 50))  # Adjusted score display position
                        
                             
                                screen.blit(instructions_text, instructions_rect)
                        
                                # Display congratulatory message and end the game when the score reaches 10
                                if score >= 10:
                                    screen.blit(congratulations_text, congratulations_rect)
                                    pygame.display.flip()
                                    pygame.time.delay(4000)  # Display the message for 4 seconds
                                    game_over = True
                        
                            # Display retry button when the game ends
                            if game_over:
                                screen.blit(retry_button_text, retry_button_rect)
                                pygame.display.flip()
                        
                                for event in pygame.event.get():
                                    if event.type == pygame.QUIT:
                                        running = False
                                    elif event.type == pygame.MOUSEBUTTONDOWN and event.button == 1:
                                
                                        if retry_button_rect.collidepoint(event.pos):
                                            reset_game()
                                            game_over = False
                        
                            pygame.display.flip()
                            clock.tick(FPS)
                        
                        
                        
                        pygame.quit()
                        sys.exit()
